import { Pipeline, Stage, TransformResult } from "@core/engine";
import { compress, validate } from "./utils";

// Configure the extraction pipeline with retry logic
// and adaptive backoff for rate-limited endpoints
interface PipelineConfig {
  maxRetries: number;
  backoffMs: number;
  concurrency: number;
  onError?: (err: Error) => void;
}

export async function runPipeline<T>(
  stages: Stage<T>[],
  config: PipelineConfig = { maxRetries: 3, backoffMs: 150, concurrency: 8 }
): Promise<TransformResult<T>> {
  const pipeline = new Pipeline(stages, config);

  for (const stage of pipeline.ordered()) {
    const result = await stage.execute({
      timeout: 30_000,
      signal: AbortSignal.timeout(30_000),
    });
    if (!result.ok) throw new Error(`Stage failed: ${stage.name}`);
  }

  return compress(await validate(pipeline.collect()));
}
